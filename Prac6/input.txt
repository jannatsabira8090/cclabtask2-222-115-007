#include <stdio.h>

#define MAX_SIZE 100

int main() {
    int numbers[MAX_SIZE], count;
    float sum = 0, average;

    // Prompt for number count
    printf("How many numbers do you want to enter (1â€“%d)? ", MAX_SIZE);
    if (scanf("%d", &count) != 1 || count <= 0 || count > MAX_SIZE) {
        printf("Invalid input. Please enter a number between 1 and %d.\n", MAX_SIZE);
        return 1;
    }

    // Input loop
    for (int i = 0; i < count; i++) {
        printf("Enter number %d: ", i + 1);
        if (scanf("%d", &numbers[i]) != 1) {
            printf("Invalid input. Please enter an integer.\n");
            // Clear input buffer
            while (getchar() != '\n');
            i--; // repeat this iteration
            continue;
        }
        sum += numbers[i];
    }

    average = sum / count;

    // Display results
    printf("\nðŸ“Š Results:\n");
    printf("Sum = %.2f\n", sum);
    printf("Average = %.2f\n", average);

    return 0;
}
